145 /*******************************************************************************
146 * The function UART_put puts a byte, to the transmit buffer at the location
147 * pointed to by tx_in_idx. The pointer is incremented circularly as described
148 * previously. If the transmit buffer should wrap around (should be designed
149 * not to happen), data will be lost. The serial interrupt must be temporarily
150 * disabled since it reads tx_in_idx and this routine updates tx_in_idx which is
151 * a 16 bit value.
152 *******************************************************************************/
153 void UART_put(UCHAR c)
154 {
155  *tx_in_ptr++ = c; /* save character to transmit buffer */
156  if( tx_in_ptr >= TX_BUF_SIZE + tx_buf)
157  tx_in_ptr = tx_buf; /* 0 <= tx_in_idx < TX_BUF_SIZE */
158 }

160 /*******************************************************************************
161 * The function UART_get gets the next byte if one is available from the receive
162 * buffer at the location pointed to by rx_out_idx. The pointer is circularly
163 * incremented and the byte is returned in R7. Should no byte be available the
164 * function will wait until one is available. There is no need to disable the
165 * serial interrupt which modifies rx_in_idx since the function is looking for a
166 * compare only between rx_in_idx & rx_out_idx.
167 *******************************************************************************/
168 UCHAR UART_get(void)
169 {
170  UCHAR c;
171  while( rx_in_ptr == rx_out_ptr ); /* wait for a received character,
172  indicated */
173  // when pointers are different
174  // this could be an infinite loop, but
175  // is not because of UART_input check
176  c = *rx_out_ptr++;
177  if( rx_out_ptr >= RX_BUF_SIZE + rx_buf ) // if at end of buffer
178  {
179  rx_out_ptr = rx_buf; /* 0 <= rx_out_idx < RX_BUF_SIZE */
180  // return byte from beginning of buffer
181  } // next time.
182  return(c);
183 }
