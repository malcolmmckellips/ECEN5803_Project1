135 uint16_t calibrateADC(void){
136  //ENABLE AND SET UP CLOCKS HERE
137 
138  //Set up hardware config for config with "highest accuracy".
139  //Set up only clock source and frequency, low power config, voltage ref, sample time, and high speed config
140  //The input channel, conversion mode cont, compare function, resolution mode, and differential/single-ended mode are all ignored
141 
142  ADC0->CFG1 &= ~(NORMAL_POWER_CONFIG << CFG1_ADLPC); //We want to be in normal power mode (0)
143  ADC0->CFG1 |= (LONG_SAMPLE << CFG1_ADLSMP); //Long samples
144  ADC0->CFG1 &= ~(BUS_CLK_IN << CFG1_ADICLK); //Input clock set to bus clock
145  ADC0->CFG1 &= ~(CLOCK_RATIO_1 << CFG1_ADIV); //We want ADIV clk selecct to 00 for divide ratio of 1
146 
147  //"For best results set hardware avging to a maximum: SC3[AVGE]=1 and SC3[AVGS]=11"
148  ADC0->SC3 |= (SC3_AVGE_EN << SC3_AVGE_FIELD);
149  ADC0->SC3 |= (SC3_AVGS_32samples << SC3_AVGS_FIELD);
150 
151  //To start calibration, clr ADCx_SC2[ADTRG] = 0 and set SC3[CAL] = 1
152  ADC0->SC2 &= ~(SC2_TRG_EN << SC2_TRG_FIELD); //clear adc0 sc2 trg
153  ADC0->SC3 |= (S3_CAL_EN << SC3_CAL_FIELD); //Set to start calibration
154 
155  //FROM RM pg. 495:
156  /*
157  1. Initialize or clear a 16-bit variable in RAM.
158  2. Add the plus-side calibration results CLP0, CLP1, CLP2, CLP3, CLP4, and CLPS to
159  the variable.
160  3. Divide the variable by two.
161  4. Set the MSB of the variable.
162  5. The previous two steps can be achieved by setting the carry bit, rotating to the right
163  through the carry bit on the high byte and again on the low byte.
164  6. Store the value in the plus-side gain calibration register PG.
165  7. Repeat the procedure for the minus-side gain calibration value.
166  */
167 
168  // Initialize or clear a 16-bit variable in RAM.
169  uint16_t PG_value = 0; //plus side gain register value
170 
171  //Add the plus-side calibration results CLP0, CLP1, CLP2, CLP3, CLP4, and CLPS to vthe variable.
172  PG_value = (ADC0->CLP0) + (ADC0->CLP1) + (ADC0->CLP2) + (ADC0->CLP3) + (ADC0->CLP4) + (ADC0->CLPS);
173  PG_value = PG_value / 2; //Divide the variable by two.
174  PG_value = PG_value |= 0x8000; //Set the MSB of the variable. could be 0xFF00 instead, Don't know if they mean most significant BYTE or BIT
175  ADC0->PG = PG_value;//Store the value in the plus-side gain calibration register PG.
176 
177  //Repeat he procedure for the minus cide calibration value
178  // Initialize or clear a 16-bit variable in RAM.
179  uint16_t MG_value = 0;
180 
181  //Add the plus-side calibration results CLM0, CLM1, CLP2, CLM3, CLM4, and CLMS to vthe variable.
182  MG_value = (ADC0->CLM0) + (ADC0->CLM1) + (ADC0->CLM2) + (ADC0->CLM3) + (ADC0->CLM4) + (ADC0->CLMS);
183  MG_value = MG_value / 2; //Divide the variable by two.
184  MG_value = MG_value |= 0x8000; //Set the MSB of the variable. could be 0xFF00 instead, Don't know if they mean most significant BYTE or BIT
185  ADC0->MG = MG_value;//Store the value in the minus-side gain calibration register MG.
186 
187 
188  //return whether calibration failed or not with CALF
189  uint16_t failure = 0;
190 
191  if (ADC0->SC3 & (1<<SC3_CALF_FIELD) )
192  failure = 1;
193 
194  return failure;
195 }

287  if ((positive) && (new_sample < neg_threshold)){

290  }
291 
292  //We are going from neg to pos (aka crossing positive threshold and having a positive peak)
293  else if ((!positive) && (new_sample > pos_threshold)){
294  positive = 1;
295  peaks++;
296  }

299  freq_index ++;

302  if (peaks>0)

305  freq_index = 0;
306  peaks = 0;
307  positive = 0;
